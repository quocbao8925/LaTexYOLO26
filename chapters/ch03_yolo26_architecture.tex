\section{Tổng quan kiến trúc}

YOLO26 gồm 3 phần chính: \textbf{Backbone} (trích xuất đặc trưng), \textbf{Neck} (tổng hợp đa tỷ lệ), và \textbf{Detection Head} (dự đoán box + class).

\section{Hệ thống Scale}
\label{sec:scale}

YOLO26 hỗ trợ 5 kích thước qua hai hệ số \texttt{depth} (độ sâu) và \texttt{width} (độ rộng):

\begin{table}[H]
\centering
\begin{tabular}{lcccl}
\toprule
\textbf{Scale} & \textbf{Depth} & \textbf{Width} & \textbf{Max Ch} & \textbf{Ghi chú} \\
\midrule
n (nano)   & 0.50 & 0.25 & 1024 & Nhỏ nhất, nhanh nhất \\
s (small)  & 0.50 & 0.50 & 1024 & Cân bằng tốc độ/chính xác \\
m (medium) & 0.50 & 1.00 & 512  & Trung bình \\
l (large)  & 1.00 & 1.00 & 512  & Lớn \\
x (xlarge) & 1.00 & 1.50 & 512  & Chính xác nhất \\
\bottomrule
\end{tabular}
\caption{Cấu hình Scale của YOLO26.}
\label{tab:scales}
\end{table}

\textbf{Công thức tính kênh và lặp:}
\begin{align}
    \texttt{ch}(c) &= \texttt{make\_divisible}\left(\min(c, \texttt{max\_ch}) \times \texttt{width},\; 8\right) \label{eq:ch} \\
    \texttt{rep}(n) &= \max\left(\text{round}(n \times \texttt{depth}),\; 1\right) \label{eq:rep}
\end{align}

Ví dụ scale nano: $\texttt{ch}(256) = \texttt{make\_divisible}(\min(256, 1024) \times 0.25, 8) = 64$.

b2  & C3k2   & $32\times80^2$ & $64\times80^2$ & c3k=F, $e\!=\!0.25$ & \\

\bottomrule
\end{tabular}
\caption{Kiến trúc Backbone YOLO26n (scale nano, ảnh $320\times320$).}
\label{tab:backbone}
\end{table}

\textbf{Các điểm quan trọng:}
\begin{itemize}
    \item Layer b0--b1: Conv stride-2 giảm resolution 4x, tăng channels.
    \item Layer b2, b4: C3k2 với \texttt{c3k=False} (Bottleneck đơn giản) --- feature đơn giản.
    \item Layer b6, b8: C3k2 với \texttt{c3k=True} (C3k kernel 3x3 sâu) --- feature phức tạp.
    \item Layer b9: SPPF tạo multi-scale receptive field (5x5, 9x9, 13x13).
    \item Layer b10: C2PSA thêm self-attention ở P5 ($10\times10$, N=100, O(N$^2$) chấp nhận được).
\end{itemize}

\section{Neck --- FPN + PAN}
\label{sec:neck}

Neck tổng hợp đặc trưng đa tỷ lệ bằng \textbf{Top-Down} (FPN) và \textbf{Bottom-Up} (PAN).

\subsection{Top-Down Pathway (FPN)}

Truyền semantic information từ tầng sâu (P5) xuống tầng nông (P3):

\begin{enumerate}
    \item Upsample b10 ($256 \times 10^2$) $\times 2 \to 256 \times 20^2$
    \item Concat với b6 ($128 \times 20^2$) $\to 384 \times 20^2$
    \item C3k2 $\to$ h13: $128 \times 20^2$
    \item Upsample h13 $\times 2 \to 128 \times 40^2$
    \item Concat với b4 ($128 \times 40^2$) $\to 256 \times 40^2$
    \item C3k2 $\to$ h16: $64 \times 40^2$ \quad (P3 output)
\end{enumerate}

\subsection{Bottom-Up Pathway (PAN)}

Truyền detail information từ tầng nông (P3) lên tầng sâu (P5):

\begin{enumerate}
    \item Conv stride-2: h16 ($64 \times 40^2$) $\to 64 \times 20^2$
    \item Concat với h13 ($128 \times 20^2$) $\to 192 \times 20^2$
    \item C3k2 $\to$ h19: $128 \times 20^2$ \quad (P4 output)
    \item Conv stride-2: h19 ($128 \times 20^2$) $\to 128 \times 10^2$
    \item Concat với b10 ($256 \times 10^2$) $\to 384 \times 10^2$
    \item C3k2(attn=True) $\to$ h22: $256 \times 10^2$ \quad (P5 output)
\end{enumerate}

\begin{notebox}[Tại sao cần cả hai chiều?]
\begin{itemize}
    \item \textbf{Top-Down ($\downarrow$)}: P3 nhận semantic từ P5 $\to$ phát hiện vật nhỏ với ngữ cảnh đầy đủ.
    \item \textbf{Bottom-Up ($\uparrow$)}: P5 nhận detail từ P3 $\to$ định vị chính xác hơn cho vật lớn.
\end{itemize}
\end{notebox}

\subsection{Upsample --- Nearest Neighbor}

\begin{equation}
    \text{Upsample}_{2\times}: [B, C, H, W] \to [B, C, 2H, 2W]
\end{equation}

Mode \texttt{nearest}: mỗi pixel được nhân đôi theo cả 2 chiều. \textbf{Không có tham số học} (zero learnable params), nhanh.

\subsection{Concat}

Ghép nối theo chiều channel (dim=1):
\begin{equation}
    \text{Concat}: [B, C_1, H, W] \oplus [B, C_2, H, W] \to [B, C_1+C_2, H, W]
\end{equation}
\textbf{Yêu cầu}: $H, W$ phải bằng nhau. Đó là lý do phải Upsample trước khi Concat.

\section{Detection Head --- Dual Head}
\label{sec:detect_head}

Detection Head nhận 3 feature maps (h16/P3, h19/P4, h22/P5) và tạo dự đoán.

\subsection{Box Head (cv2)}

\begin{equation}
    \text{cv2}: \text{Conv}(x, c_2, 3) \to \text{Conv}(c_2, c_2, 3) \to \text{Conv2d}(c_2, 4 \times \texttt{reg\_max}, 1)
\end{equation}

Với \texttt{reg\_max=1}: output = 4 channels $(l, t, r, b)$ trực tiếp.

\subsection{Classification Head (cv3)}

\begin{equation}
    \text{cv3}: [\text{DWConv}(x) + \text{Conv}] \to [\text{DWConv} + \text{Conv}] \to \text{Conv2d}(c_3, nc, 1)
\end{equation}

DWConv giảm tính toán so với Conv tiêu chuẩn.

\subsection{One-to-Many (O2M) Head}

Mỗi ground truth box gán cho \textbf{nhiều} anchor predictions (topk=10). Cung cấp dense supervision, \textbf{chỉ dùng khi training}.

\subsection{One-to-One (O2O) Head}

\begin{itemize}
    \item Deep copy từ O2M: cùng kiến trúc, trọng số khởi tạo giống nhau.
    \item Mỗi GT chỉ gán cho \textbf{đúng 1} anchor tốt nhất (topk=7, topk2=1).
    \item Nhận \textbf{detached features} ($x.\text{detach}()$) --- gradient O2O không chảy về backbone.
    \item Dùng cho cả training và inference --- \textbf{loại bỏ hoàn toàn NMS}.
\end{itemize}

\subsection{Giải mã Bounding Box}

Từ dự đoán $(l, t, r, b)$ và anchor point $(a_x, a_y)$:
\begin{equation}
    \text{bbox}_{\text{xyxy}} = (a_x - l, \; a_y - t, \; a_x + r, \; a_y + b) \times \text{stride}
\end{equation}

\subsection{Postprocessing (Inference)}

\begin{enumerate}
    \item Chỉ chạy O2O head (bỏ O2M)
    \item Tính confidence: $\text{score} = \sigma(\text{cls\_logits})$
    \item Chọn Top-K predictions (mặc định $K=300$)
    \item Output: $(x_1, y_1, x_2, y_2, \text{score}, \text{class\_id})$
\end{enumerate}

Khi export cho inference, O2M head (cv2, cv3) bị loại bỏ:

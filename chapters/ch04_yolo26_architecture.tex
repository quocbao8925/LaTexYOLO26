\chapter{Kiến trúc mô hình YOLO26}
\label{ch:architecture}

Chương này trình bày chi tiết kiến trúc mô hình \yolo, bao gồm ba phần chính: Backbone (trích xuất đặc trưng), Neck (tổng hợp đặc trưng đa tỷ lệ), và Detection Head (dự đoán bounding box và lớp đối tượng). \section{Backbone}
\label{sec:arch_backbone}

Backbone của \yolo gồm 11 lớp (b0--b10) có nhiệm vụ trích xuất đặc trưng ảnh ở nhiều mức phân giải khác nhau.

\subsection{Khối Conv cơ bản}

Trong \yolo, mỗi khối \texttt{Conv} bao gồm ba phần: lớp tích chập (\texttt{Conv2d}), chuẩn hóa batch (\texttt{BatchNorm2d}), và hàm kích hoạt \texttt{SiLU}:

\begin{lstlisting}[caption={Khối Conv cơ bản trong YOLO26}, label=lst:conv]
class Conv(nn.Module):
    def __init__(self, c1, c2, k=1, s=1, p=None, g=1):
        self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p), groups=g, bias=False)
        self.bn = nn.BatchNorm2d(c2)
        self.act = nn.SiLU(inplace=True)
    
    def forward(self, x):
        return self.act(self.bn(self.conv(x)))
\end{lstlisting}

Các lớp Conv đầu tiên (b0, b1, b3, b5, b7) sử dụng kernel $3 \times 3$ với stride $s=2$ để giảm kích thước không gian xuống một nửa ở mỗi bước.

\textbf{Conv 1x1} trộn channel tại cùng vị trí spatial. \textbf{Conv 3x3} trộn thông tin không gian (cạnh, biên, texture).

\subsection{Khối DWConv (Depthwise Convolution)}
\label{subsec:dwconv}

DWConv là tích chập tách biệt theo chiều sâu (Depthwise Convolution), trong đó số lượng nhóm tính toán bằng số kênh vào và số kênh ra ($\text{groups} = \gcd(c_1, c_2)$):
\begin{equation}
    \text{Params}_{\text{DW}} = c_1 \times k^2 \quad \ll \quad \text{Params}_{\text{std}} = c_1 \times c_2 \times k^2
\end{equation}
DWConv được sử dụng rộng rãi trong classification head (cv3) của YOLO26 nhằm giảm chi phí tính toán và tăng tốc độ suy luận.

\subsection{Khối Bottleneck}
\label{subsec:bottleneck}

Bottleneck là block residual cơ bản, tạo đặc trưng $F(x)$ bằng 2 lớp Conv rồi tùy điều kiện cộng residual với đầu vào $x$:

\begin{figure}[H]
\centering
\includegraphics[width=0.45\textwidth]{bottleneck_block.png}
\caption{Sơ đồ khối Bottleneck với skip connection.}
\label{fig:bottleneck}
\end{figure}

\begin{equation}
    F(x) = \text{cv2}(\text{cv1}(x))
    \label{eq:bottleneck_Fx}
\end{equation}
\begin{equation}
    \text{Output} = \begin{cases}
        x + F(x) & \text{nếu } \texttt{shortcut} = \texttt{True} \text{ và } c_1 = c_2 \\
        F(x) & \text{ngược lại}
    \end{cases}
    \label{eq:bottleneck_output}
\end{equation}

\noindent trong đó:
\begin{itemize}
    \item \texttt{cv1}: Conv$(c_1, c\_, k_1, s=1)$ với $c\_ = \lfloor c_2 \times e \rfloor$ (kênh ẩn, $e$ là expansion ratio).
    \item \texttt{cv2}: Conv$(c\_, c_2, k_2, s=1)$ với $\texttt{groups}=g$.
    \item Nhánh skip chỉ kích hoạt khi $c_1 = c_2$ và \texttt{shortcut=True}.
\end{itemize}

\textbf{Tác dụng}: Giảm chi phí tính toán (nhờ expansion ratio $e < 1$), cho phép xây dựng mạng sâu hơn mà vẫn nhẹ, và giữ thông tin gốc nhờ nhánh residual giúp tránh vanishing gradient.

Ví dụ: input $[B, 128, 80, 80]$, $e=0.5$ $\to$ $c\_=64$ $\to$ output $[B, 128, 80, 80]$ (chiều dài và chiều rộng giữ nguyên).

\subsection{Khối C2f (Cross Stage Partial with 2 Convolutions)}
\label{subsec:c2f}

Khối C2f~\cite{cspnet} kết hợp tư tưởng của CSPNet nhằm gia tăng số lượng đường truyền gradient, giúp mạng học được nhiều đặc trưng đa dạng hơn:

\begin{equation}
    \text{C2f}(x) = \text{cv2}\Big(\text{Concat}\big[y_0, y_1, m_1(y_1), m_2(m_1(y_1)), \ldots\big]\Big)
\end{equation}

Đầu vào sau khi qua \texttt{cv1} được chia đôi thành 2 nhánh: $[y_0, y_1] = \text{chunk}(\text{cv1}(x), 2)$. Nhánh nông $y_0$ đi thẳng không xử lý sâu thêm, trong khi nhánh $y_1$ được truyền qua một chuỗi các khối Bottleneck $m_i$.

\textbf{Sơ đồ luồng dữ liệu (với n=2)}:
\begin{verbatim}
Input -> cv1 -> split -> [y0, y1]
                              |
                         Bottleneck(y1) -> y2
                              |
                         Bottleneck(y2) -> y3
                              |
              cat(y0, y1, y2, y3) -> cv2 -> Output
\end{verbatim}

\textbf{Ý nghĩa}: Cơ chế split-and-concat này vừa giữ lại các đặc trưng nông thô ($y_0$) vừa trích xuất các đặc trưng ngữ cảnh sâu ($y_3$), giúp gradient lan truyền hiệu quả hơn trong quá trình backward.

\subsection{Khối C3 và C3k}
\label{subsec:c3_c3k}

\textbf{C3} là khối CSP kiểu cũ có hai nhánh chính: nhánh xử lý sâu qua chuỗi Bottleneck và nhánh đi thẳng qua một khối Conv, sau đó ghép lại bằng phép concat và xử lý qua Conv 1x1 cuối.

\textbf{C3k} kế thừa C3 nhưng sử dụng Bottleneck với kernel $k \times k$ tùy chỉnh (thường $k=3$):
\begin{equation}
    \text{C3k}: \text{Bottleneck}_{\text{C3}} \text{ dùng } k=((1,1),(3,3)) \quad \to \quad \text{C3k dùng } k=(3,3)
\end{equation}

\begin{figure}[H]
\centering
\includegraphics[width=0.55\textwidth]{c3k_block.png}
\caption{Sơ đồ khối C3k --- C3 với kernel 3x3 sâu hơn.}
\label{fig:c3k}
\end{figure}

Khối C3k có khả năng bao quát thông tin không gian tốt hơn nhờ mở rộng kernel tích chập ở nhánh sâu, giúp học được các đặc trưng biên và cấu trúc hình học tốt hơn, đổi lại chi phí tính toán cao hơn Bottleneck thông thường.

\subsection{Khối C3k2}
\label{subsec:c3k2}

C3k2 là khối trộn đặc trưng chính và quan trọng nhất trong \yolo, kế thừa từ C2f nhưng cho phép tùy biến loại block bên trong:

\begin{equation}
    y_0, y_1 = \text{Split}(\text{cv1}(x))
    \label{eq:c3k2_split}
\end{equation}
\begin{equation}
    z_i = \text{Block}_i(z_{i-1}), \quad z_0 = y_1, \quad i = 1, \ldots, n
    \label{eq:c3k2_blocks}
\end{equation}
\begin{equation}
    \text{Output} = \text{cv2}\big(\text{cat}([y_0, y_1, z_1, \ldots, z_n])\big)
    \label{eq:c3k2_output}
\end{equation}

\begin{figure}[H]
\centering
\includegraphics[width=0.55\textwidth]{c3k2_block.png}
\caption{Sơ đồ khối C3k2 --- forward theo C2f, block bên trong tùy cấu hình.}
\label{fig:c3k2}
\end{figure}

Tùy vào 2 tham số boolean \texttt{attn} và \texttt{c3k}, các Block bên trong sẽ được khởi tạo khác nhau:
\begin{enumerate}
    \item \texttt{attn=True}: Block = Bottleneck + PSABlock (Position-Sensitive Attention).
    \item \texttt{attn=False, c3k=True}: Block = C3k (kernel $k \times k$ sâu).
    \item \texttt{attn=False, c3k=False}: Block = Bottleneck tiêu chuẩn.
\end{enumerate}

\begin{table}[H]
\centering
\small
\caption{Sử dụng C3k2 tại các vị trí trong YOLO26.}
\label{tab:c3k2_usage}
\begin{tabular}{lcccl}
\toprule
\textbf{Layer} & \textbf{c3k} & \textbf{attn} & \textbf{Module bên trong} & \textbf{Vị trí} \\
\midrule
b2, b4 & False & False & Bottleneck đơn giản & Backbone đầu \\
b6, b8 & True & False & C3k (3x3 sâu) & Backbone P4, P5 \\
h13--h19 & True & False & C3k & Neck \\
h22 & True & True & Bottleneck + PSABlock & Neck P5 output \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Ví dụ sự thay đổi kích thước tensor}: C3k2($c_1=1024, c_2=256, n=2, \text{c3k}=\text{True}, e=0.5$):
\begin{verbatim}
[B, 1024, 80, 80] -> cv1 -> [B, 256, 80, 80]
    -> split: y0 [B, 128, 80, 80], y1 [B, 128, 80, 80]
    -> C3k(y1) -> y2 [B, 128, 80, 80]
    -> C3k(y2) -> y3 [B, 128, 80, 80]
    -> cat(y0,y1,y2,y3) -> [B, 512, 80, 80]
    -> cv2 -> [B, 256, 80, 80]
\end{verbatim}
Kích thước không gian $H, W$ được giữ nguyên hoàn toàn do tất cả các lớp tích chập bên trong đều sử dụng stride=1 và cơ chế tự động đệm (autopad).

\subsection{Khối SPPF (Spatial Pyramid Pooling Fast)}
\label{subsec:sppf}

SPPF mở rộng vùng nhận cảm ngữ cảnh (receptive field) bằng cách áp dụng liên tiếp 3 lớp MaxPool $5 \times 5$ song song với việc lưu lại kết quả trung gian, mà không làm giảm kích thước không gian ảnh đầu vào:

\begin{figure}[H]
\centering
\includegraphics[width=0.55\textwidth]{sppf_block.png}
\caption{Sơ đồ khối SPPF --- MaxPool liên tiếp tạo multi-scale receptive field.}
\label{fig:sppf}
\end{figure}

\begin{equation}
    y_0 = \text{cv1}(x), \quad
    y_1 = \text{MaxPool}(y_0), \quad
    y_2 = \text{MaxPool}(y_1), \quad
    y_3 = \text{MaxPool}(y_2)
    \label{eq:sppf}
\end{equation}
\begin{equation}
    \text{Output} = \text{cv2}\big(\text{cat}([y_0, y_1, y_2, y_3])\big)
\end{equation}

\begin{figure}[H]
\centering
\includegraphics[width=0.75\textwidth]{sppf_maxpool_matrices.png}
\caption{Minh họa MaxPool 5x5 qua 3 lần liên tiếp --- tạo các receptive field có kích thước tương đương $5\times5$, $9\times9$, và $13\times13$.}
\label{fig:sppf_maxpool}
\end{figure}

Mỗi lần MaxPool $5 \times 5$ (với stride $s=1$, padding $p=2$) mở rộng vùng ngữ cảnh thêm 4 pixel. Qua 3 lần lặp, thông tin ngữ cảnh toàn cục được tổng hợp tương đương một cửa sổ MaxPool có kích thước $13 \times 13$, hỗ trợ mô hình nhận diện vật thể ở nhiều kích thước khác nhau.

\textbf{So với SPP}: SPP dùng 3 kernel song song ($5\times5$, $9\times9$, $13\times13$); trong khi SPPF dùng cùng một kernel $5\times5$ nối tiếp --- cho receptive field tương đương nhưng \textbf{giảm đáng kể chi phí tính toán}.

\begin{notebox}[Lưu ý: YOLO26 SPPF có act=False trên cv1]
Khác với SPPF gốc (Ultralytics) dùng \texttt{act=True} (SiLU) trên lớp tích chập cv1, YOLO26 sử dụng \texttt{act=False}. Thiết kế này làm giảm dung lượng biểu diễn phi tuyến tính ngay trước bước max-pooling, giúp ổn định thông tin biên của vật thể lớn.
\end{notebox}

\subsection{Cơ chế Attention (Multi-Head Self-Attention)}
\label{subsec:attention}

Trong các cấu hình nâng cao của YOLO26, Multi-Head Self-Attention được chèn vào Neck (h22) và Backbone (b10) để bắt cặp các tương quan không gian toàn cục:

\begin{figure}[H]
\centering
\includegraphics[width=0.55\textwidth]{attention_block.png}
\caption{Sơ đồ khối Attention với QKV projection và Position Encoding.}
\label{fig:attention}
\end{figure}

\begin{align}
    Q, K, V &= \text{split}(\text{QKV\_Conv}(x)) \\
    \text{Attn} &= \text{softmax}\left(\frac{Q \cdot K^T}{\sqrt{d_k}}\right) \\
    \text{Output} &= V \cdot \text{Attn}^T + \text{PE}(V)
\end{align}

\begin{itemize}
    \item \textbf{QKV Conv}: Lớp tích chập Conv $1\times1$ chiếu đồng thời vector đầu vào sang 3 không gian Query ($Q$), Key ($K$), Value ($V$). Lớp này không sử dụng activation.
    \item $d_k = \texttt{head\_dim} \times \texttt{attn\_ratio}$: Kích thước vector Key được nén nhỏ để tiết kiệm bộ nhớ.
    \item \textbf{PE (Position Encoding)}: Sử dụng một khối Depthwise Conv $3\times3$ tác động lên Value ($V$) nhằm bảo toàn thông tin cấu trúc không gian hình học vốn bị mất trong cơ chế tính toán ma trận Attention thông thường.
    \item \textbf{Multi-head}: Kênh được chia thành $h$ heads riêng biệt để xử lý song song các vùng đặc trưng độc lập.
\end{itemize}

\begin{figure}[H]
\centering
\includegraphics[width=0.85\textwidth]{attention_flow_shapes_matrices.png}
\caption{Chi tiết luồng xử lý và sự biến đổi kích thước tensor trong module Attention.}
\label{fig:attention_flow}
\end{figure}

\subsection{Khối PSABlock (Position-Sensitive Attention Block)}
\label{subsec:psablock}

PSABlock tích hợp cơ chế Self-Attention bên trên và một khối Feed-Forward Network (FFN) theo dạng nối tiếp, đi kèm hai kết nối tắt (residual connection):

\begin{figure}[H]
\centering
\includegraphics[width=0.45\textwidth]{psablock_block.png}
\caption{Sơ đồ PSABlock: Attention + FFN với các kết nối residual.}
\label{fig:psablock}
\end{figure}

\begin{align}
    x_1 &= x + \text{Attention}(x) \quad \text{(Residual Attention)} \\
    x_2 &= x_1 + \text{FFN}(x_1) \quad \text{(Residual FFN)}
\end{align}

FFN (Feed-Forward Network) gồm lớp Conv $1\times1$ tăng số kênh ($c \to 2c$), đi qua hàm kích hoạt SiLU, và lớp Conv $1\times1$ nén kênh ngược về ban đầu ($2c \to c$, act=False):
\begin{equation}
    \text{FFN}(x_1) = \text{cv2}\Big(\text{SiLU}\big(\text{cv1}(x_1)\big)\Big)
\end{equation}

\subsection{Khối C2PSA (Channel-Split Position-Sensitive Attention)}
\label{subsec:c2psa}

Khối C2PSA áp dụng cơ chế chia nhánh (Channel-Split) của CSPNet kết hợp với PSABlock nhằm tối ưu hóa chi phí tính toán của cơ chế Self-Attention:

\begin{figure}[H]
\centering
\includegraphics[width=0.45\textwidth]{c2psa_block.png}
\caption{Sơ đồ C2PSA: CSP splitting + PSABlock.}
\label{fig:c2psa}
\end{figure}

\begin{align}
    a, b &= \text{split}(\text{cv1}(x)) \\
    b' &= \text{PSABlock}(b) \\
    \text{Output} &= \text{cv2}(\text{Concat}[a, b'])
\end{align}

Đầu vào qua \texttt{cv1} được phân đôi: nhánh $a$ đi thẳng bảo toàn đặc trưng tích chập cục bộ, nhánh $b$ đi qua PSABlock học các phụ thuộc không gian xa. Cuối cùng, hai nhánh được concat và hợp nhất qua \texttt{cv2}.

\begin{figure}[H]
\centering
\includegraphics[width=0.85\textwidth]{sppf_c2psa_featuremap_transform.png}
\caption{Mô phỏng sự biến đổi của feature map khi đi qua SPPF và C2PSA.}
\label{fig:sppf_c2psa_transform}
\end{figure}

\begin{notebox}[Tại sao chỉ dùng Attention ở P5?]
Độ phức tạp tính toán của cơ chế Self-Attention là $O(N^2)$ với $N = H \times W$ là số điểm ảnh. Ở tầng P5 ($10 \times 10$), $N=100$ nên chi phí cực kỳ nhẹ. Ngược lại, ở tầng P3 ($40 \times 40$), $N=1600$ khiến độ phức tạp tăng gấp 256 lần, quá nặng cho các mô hình nhỏ như YOLO26-Nano.
\end{notebox}

\section{Neck}
\label{sec:arch_neck}

Neck của \yolo sử dụng kiến trúc kết hợp FPN (top-down) và PAN (bottom-up) để tổng hợp đặc trưng đa tỷ lệ từ Backbone trước khi đưa vào Detection Head.

\subsection{Top-Down Pathway (FPN)}

Truyền semantic information từ tầng sâu (P5) xuống tầng nông (P3):

\begin{enumerate}
    \item Upsample b10 ($256 \times 10^2$) $\times 2 \to 256 \times 20^2$
    \item Concat với b6 ($128 \times 20^2$) $\to 384 \times 20^2$
    \item C3k2 $\to$ h13: $128 \times 20^2$
    \item Upsample h13 $\times 2 \to 128 \times 40^2$
    \item Concat với b4 ($128 \times 40^2$) $\to 256 \times 40^2$

\end{enumerate}

\label{sec:arch_head}

\subsection{Nhánh cv2 --- Box Prediction}

Dự đoán tọa độ bounding box thông qua 2 lớp Conv $3 \times 3$ và một lớp Conv $1 \times 1$ cuối:

\begin{lstlisting}[caption={Nhánh cv2 (Box) trong Detection Head}, label=lst:cv2]
self.cv2 = nn.ModuleList(
    nn.Sequential(
        Conv(x, c2, 3),                     # Conv 3x3
        Conv(c2, c2, 3),                    # Conv 3x3
        nn.Conv2d(c2, 4 * self.reg_max, 1)  # Conv 1x1 -> 4*reg_max
    ) for x in ch
)
\end{lstlisting}

Đầu ra: $4 \times \text{reg\_max}$ kênh. Với Direct Regression ($\text{reg\_max} = 1$), đầu ra chỉ có 4 kênh tương ứng 4 khoảng cách LTRB (left, top, right, bottom).

        nn.Sequential(DWConv(x, x, 3), Conv(x, c3, 1)),  # DWConv + Conv
        nn.Sequential(DWConv(c3, c3, 3), Conv(c3, c3, 1)),
        nn.Conv2d(c3, self.nc, 1)  # Conv 1x1 -> nc classes
    ) for x in ch
)
\end{lstlisting}

\section{Box Decoding: Direct Regression}
\label{sec:arch_box_decoding}

\yolo sử dụng phương pháp hồi quy trực tiếp (Direct Regression) thay cho DFL~\cite{dfl}, giúp giảm đáng kể khối lượng tính toán:

\begin{enumerate}[label=\textbf{Bước \arabic*:}]
    \item \textbf{Feature Extraction}: Trích xuất đặc trưng qua Backbone và Neck.
    \item \textbf{Tạo Anchor Grids}: Xác định tọa độ tâm $(c_x, c_y)$ cho mỗi ô lưới.
    \item \textbf{Dự đoán LTRB}: Nhánh cv2 xuất ra 4 giá trị khoảng cách: $d_\text{left}$, $d_\text{top}$, $d_\text{right}$, $d_\text{bottom}$.
    \item \textbf{Khôi phục tọa độ}:
    \begin{equation}
        x_\text{min} = c_x - d_\text{left}, \quad y_\text{min} = c_y - d_\text{top}
    \end{equation}
    \begin{equation}
        x_\text{max} = c_x + d_\text{right}, \quad y_\text{max} = c_y + d_\text{bottom}
    \end{equation}
    \item \textbf{Rescale}: Nhân tọa độ với stride tương ứng để đưa về kích thước ảnh gốc.
\end{enumerate}

\subsection{Cơ chế DFL (Distribution Focal Loss)}
\label{subsec:dfl_module}

DFL~\cite{dfl} chuyển bài toán hồi quy khoảng cách liên tục thành bài toán phân loại trên một số lượng $\texttt{reg\_max}$ các bins rời rạc:

\begin{equation}
    \hat{d} = \sum_{i=0}^{\texttt{reg\_max}-1} i \cdot p_i, \quad p_i = \text{softmax}(\text{logits})_i
\end{equation}

Trọng số của lớp Conv $1\times1$ cuối được cố định là chuỗi tăng dần $[0, 1, 2, \ldots, \texttt{reg\_max}-1]$ và không tham gia vào quá trình lan truyền ngược để cập nhật tham số.

\begin{notebox}[YOLO26 mặc định: reg\_max=1 --- không dùng DFL]
Khi đặt cấu hình mặc định \texttt{reg\_max=1}, DFL sẽ tự động được thay thế bằng một lớp đồng nhất \texttt{nn.Identity()}. Lúc này mô hình thực hiện hồi quy trực tiếp 4 giá trị khoảng cách $(l, t, r, b)$, đầu ra của box head chỉ gồm 4 kênh. Với cấu hình \texttt{reg\_max=16}, đầu ra sẽ có 64 kênh, cho phép dự đoán phân phối xác suất biên tốt hơn nhưng tăng chi phí tính toán.
\end{notebox}

\subsection{So sánh Direct Regression và DFL}

\begin{table}[H]
    \centering

\textbf{Công thức tính kênh và lặp:}
\begin{align}
    \texttt{ch}(c) &= \texttt{make\_divisible}\left(\min(c, \texttt{max\_ch}) \times \texttt{width},\; 8\right) \label{eq:ch} \\
    \texttt{rep}(n) &= \max\left(\text{round}(n \times \texttt{depth}),\; 1\right) \label{eq:rep}
\end{align}

Ví dụ scale nano: $\texttt{ch}(256) = \texttt{make\_divisible}(\min(256, 1024) \times 0.25, 8) = 64$.

Trong đồ án, nhóm sử dụng biến thể \textbf{nano} (n) với tổng dung lượng weights khoảng 5.3MB, phù hợp cho triển khai trên \rpi.

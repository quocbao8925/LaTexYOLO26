\section{Tổng quan Loss trong YOLO26}

Tổng loss gồm 3 thành phần:
\begin{equation}
    \mathcal{L}_{\text{head}} = w_{\text{box}} \cdot \mathcal{L}_{\text{box}} + w_{\text{cls}} \cdot \mathcal{L}_{\text{cls}} + w_{\text{dfl}} \cdot \mathcal{L}_{\text{dfl}}
    \label{eq:total_loss}
\end{equation}

\section{Box Loss --- CIoU Loss}

\begin{equation}
    \mathcal{L}_{\text{box}} = \frac{\sum_{i \in \text{fg}} (1 - \text{CIoU}_i) \cdot w_i}{\sum_j t_j}
\end{equation}
$w_i = \sum_c t_{i,c}$ là tổng target scores của anchor $i$ (chỉ foreground), $t_j$ = target\_scores\_sum.

\textbf{CIoU} (xem \secref{sec:ciou}) xét cả khoảng cách tâm, diện tích giao, và tỷ lệ khung hình --- gradient tốt hơn IoU thuần.

% ---------------------------------------------------------------
\section{Classification Loss --- BCE Loss}

\begin{equation}
    \mathcal{L}_{\text{cls}} = \frac{\sum_{i,c} \text{BCE}(\hat{s}_{i,c}, \; t_{i,c})}{\sum_j t_j}
\end{equation}
Target scores $t_{i,c}$ là \textbf{soft labels} từ alignment metric (không phải 0/1 cứng), giúp model học ``mức độ phù hợp'' thay vì chỉ binary.

% ---------------------------------------------------------------
\section{DFL/LTRB Loss}
\label{sec:dfl_loss}

\subsection{Khi reg\_max $>$ 1: DFL Loss}

\begin{equation}
    \mathcal{L}_{\text{DFL}}(d) = -\Big[(d_{\lceil} - d) \log p_{d_{\lfloor}} + (d - d_{\lfloor}) \log p_{d_{\lceil}}\Big]
\end{equation}
$d$ = target liên tục, $d_{\lfloor} = \lfloor d \rfloor$, $d_{\lceil} = d_{\lfloor} + 1$, $p_i = \text{softmax}(\text{logits})_i$.

\subsection{Khi reg\_max = 1: L1 Loss (YOLO26 default)}

\begin{equation}
    \mathcal{L}_{\text{L1}} = \frac{\sum_{i \in \text{fg}} \|p_i - t_i\|_1 \cdot w_i}{\sum_j t_j}
\end{equation}
Trong đó $p_i, t_i$ là predicted/target LTRB distances, được normalize theo image size:
\begin{align}
    t_{\text{ltrb}}[0::2] &\mathrel{/}= \text{width}, \quad t_{\text{ltrb}}[1::2] \mathrel{/}= \text{height}
\end{align}

\section{TaskAlignedAssigner (TAL)}
\label{sec:tal}

TAL xác định anchor nào là positive/negative cho mỗi GT box.

\subsection{Alignment Metric}

\begin{equation}
    \text{align\_metric}_{i,j} = s_{i,j}^{\alpha} \cdot u_{i,j}^{\beta}
    \label{eq:align_metric}
\end{equation}

\begin{itemize}
    \item $s_{i,j}$: classification score (sigmoid) của anchor $i$ cho class GT $j$
    \item $u_{i,j}$: CIoU giữa predicted box và GT box
    \item $\alpha = 0.5$, $\beta = 6.0$: CIoU được nhấn mạnh rất mạnh ($\beta = 6$)
\end{itemize}

\subsection{Quy trình gán nhãn}

\begin{algorithm}[H]
\caption{TaskAlignedAssigner}
\label{alg:tal}
\begin{algorithmic}[1]
\State \textbf{Input}: pred\_scores, pred\_bboxes, anchor\_points, gt\_labels, gt\_bboxes
\State \textbf{Step 1}: Lọc anchors nằm trong GT box (geometric prior)
\State \textbf{Step 2}: Tính alignment\_metric = $s^{0.5} \cdot \text{CIoU}^{6}$
\State \textbf{Step 3}: Chọn Top-K anchors có metric cao nhất cho mỗi GT
\Statex \quad\quad O2M: $K=10$, O2O: $K=7 \to$ chọn Top-1
\State \textbf{Step 4}: Xử lý conflict: nếu anchor match nhiều GT, giữ GT có IoU cao nhất
\State \textbf{Step 5}: Tính soft labels = one-hot $\times$ normalized metric
\State \textbf{Output}: target\_labels, target\_bboxes, target\_scores, fg\_mask
\end{algorithmic}
\end{algorithm}

\begin{notebox}[Anchor gần tâm không chắc được chọn]
Vì $\beta = 6.0$, CIoU ảnh hưởng rất mạnh. Anchor lệch tâm nhưng có pred\_box khớp GT hơn \textit{và} class score tốt hơn có thể được ưu tiên hơn anchor ngay tâm.
\end{notebox}

\subsection{STAL --- Small-Target-Aware Label Assignment}

Đối tượng nhỏ hơn stride$[0]$ pixel có nguy cơ không match anchor nào. STAL phóng to bbox:
\begin{equation}
    \text{bbox}'_{w,h} = \begin{cases}
    \text{stride}[1] & \text{nếu } w \text{ hoặc } h < \text{stride}[0] \\
    w, h & \text{ngược lại}
    \end{cases}
\end{equation}

\section{E2ELoss --- End-to-End Loss}
\label{sec:e2e_loss}

E2ELoss kết hợp loss từ cả O2M và O2O head:
\begin{equation}
    \mathcal{L}_{\text{E2E}} = \alpha(e) \cdot \mathcal{L}_{\text{O2M}} + \beta(e) \cdot \mathcal{L}_{\text{O2O}}, \quad \alpha + \beta = 1
    \label{eq:e2e_loss}
\end{equation}

Trọng số thay đổi theo epoch:
\begin{align}
    \alpha(e) &= \max\left(1 - \frac{e}{\max(E-1, 1)}, 0\right) \times (\alpha_0 - \alpha_f) + \alpha_f \\
    \beta(e) &= 1 - \alpha(e)
\end{align}
$\alpha_0 = 0.8$, $\alpha_f = 0.3$.

\midrule

50  & 0.55 & 0.45 & Cân bằng dần \\
100 & 0.30 & 0.70 & O2O tiếp quản --- chuyên cho inference \\
\bottomrule

\section{ProgLoss --- Progressive Loss Balancing}
\label{sec:progloss}

ProgLoss thay đổi trọng số box/cls loss theo thời gian huấn luyện:

\begin{align}
    t &= \min\left(\frac{\text{epoch}}{\text{total\_epochs} - 1}, \; 1\right) \\
    \text{progress} &= \frac{1}{2}\left(1 - \cos(\pi \cdot t)\right) \quad \text{(cosine annealing)} \label{eq:cosine_prog} \\
    w_{\text{box}}(e) &= 7.5 + (3.75 - 7.5) \times \text{progress} \\
    w_{\text{cls}}(e) &= 0.5 + (1.5 - 0.5) \times \text{progress}
\end{align}

\begin{figure}[H]
\centering
\includegraphics[width=\textwidth]{progloss_curves.png}
\caption{ProgLoss curves. Trái: O2M/O2O weights. Phải: Box weight giảm, Cls weight tăng.}
\label{fig:progloss}
\end{figure}

\textbf{Triết lý}: Giai đoạn đầu tập trung học \textit{ở đâu} (box weight cao), giai đoạn sau tập trung học \textit{là gì} (cls weight cao).

\textbf{Cosine vs Linear}: Cosine gần như không đổi ở 20 epoch đầu (ổn định), thay đổi mạnh ở giữa, nhẹ nhàng hội tụ cuối. Linear thay đổi đều, có thể gây gradient shock.

\section{Công thức ghép đầy đủ}

\begin{equation}
\boxed{
\mathcal{L}_{\text{total}}(e) = \alpha(e) \cdot \Big[w_{\text{box}}(e) \cdot \mathcal{L}_{\text{box}}^{\text{O2M}} + w_{\text{cls}}(e) \cdot \mathcal{L}_{\text{cls}}^{\text{O2M}} + w_{\text{dfl}} \cdot \mathcal{L}_{\text{dfl}}^{\text{O2M}}\Big]
+ \beta(e) \cdot \Big[w_{\text{box}}(e) \cdot \mathcal{L}_{\text{box}}^{\text{O2O}} + w_{\text{cls}}(e) \cdot \mathcal{L}_{\text{cls}}^{\text{O2O}} + w_{\text{dfl}} \cdot \mathcal{L}_{\text{dfl}}^{\text{O2O}}\Big]
}
\label{eq:full_loss}
\end{equation}

\textbf{Trực giác}:
\begin{itemize}
    \item Đầu training: $\alpha$ cao $\to$ nhiều anchor positive (dense supervision); $w_{\text{box}}$ cao $\to$ ưu tiên localization.
    \item Cuối training: $\beta$ cao $\to$ O2O head quan trọng hơn; $w_{\text{cls}}$ cao $\to$ tập trung classification.
\end{itemize}

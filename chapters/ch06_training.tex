\section{Training Loop}

Training loop chuẩn PyTorch:
\begin{verbatim}
for epoch in range(epochs):
    scheduler.step()
    model.train()
    for batch in train_loader:
        imgs = batch["img"].cuda().float() / 255
        with torch.autocast("cuda"):
            preds = model(imgs)
            loss = criterion(preds, batch)
        scaler.scale(loss).backward()
        scaler.unscale_(optimizer)
        clip_grad_norm_(model.parameters(), 10.0)
        scaler.step(optimizer); scaler.update()
        optimizer.zero_grad()
        ema.update(model)
    criterion.update()  # ProgLoss
    metrics = validate(ema.ema, val_loader)
    save_model(epoch, ema, optimizer, metrics)
    if stopper(epoch, fitness): break
\end{verbatim}

\section{Learning Rate Schedule}

Cosine annealing LR scheduler:
\begin{equation}
    \text{lr}(e) = \text{lr}_0 \times \left[\frac{1 - \cos(\pi \cdot e / E)}{2} \times (\text{lrf} - 1) + 1\right]
\end{equation}
$\text{lrf} = 0.01$ (tỷ lệ LR cuối/đầu). LR giảm từ $0.01$ xuống $0.0001$ theo đường cosine.

\section{Warmup}

Trong $n_w = \max(3 \times n_b, 100)$ iterations đầu:
\begin{align}
    \text{lr}: \quad &0 \to \text{lr}_0 \times \text{lf(epoch)} \\
    \text{momentum}: \quad &0.8 \to 0.937
\end{align}
Tăng tuyến tính bằng \texttt{np.interp()}.

\section{Exponential Moving Average (EMA)}
\label{sec:ema}

\begin{conceptbox}[Khái niệm: EMA]
EMA duy trì bản sao ``trung bình'' của model weights:
\begin{equation}
    \theta_{\text{EMA}}^{(t)} = d(t) \cdot \theta_{\text{EMA}}^{(t-1)} + (1 - d(t)) \cdot \theta_{\text{model}}^{(t)}
\end{equation}
\begin{equation}
    d(t) = d_0 \times \left(1 - e^{-t/\tau}\right), \quad d_0 = 0.9999, \; \tau = 2000
\end{equation}
EMA cho kết quả ổn định hơn, được dùng cho validation và lưu checkpoint.
\end{conceptbox}

\section{Automatic Mixed Precision (AMP)}

AMP sử dụng FP16 cho forward pass (tiết kiệm bộ nhớ, tăng tốc), FP32 cho gradient accumulation (ổn định số học):
\begin{itemize}
    \item \texttt{torch.autocast("cuda")}: Tự chọn FP16/FP32 cho từng operation
    \item \texttt{GradScaler}: Scale loss trước backward, unscale gradient trước optimizer step
\end{itemize}

\section{Gradient Clipping}

Giới hạn norm gradient để tránh exploding gradient:
\begin{equation}
    \text{clip\_grad\_norm\_}(\theta, \text{max\_norm}=10.0)
\end{equation}

% ---------------------------------------------------------------
\section{Gradient Accumulation}

Khi batch size nhỏ, gradient tích lũy qua nhiều bước:
\begin{equation}
    \text{accumulate} = \max\left(\text{round}\left(\frac{64}{\text{batch\_size}}\right), 1\right)
\end{equation}

\section{EarlyStopping}

Dừng training khi fitness không cải thiện sau \texttt{patience} epochs:
\begin{equation}
    \text{stop} = (\text{epoch} - \text{best\_epoch}) \geq \text{patience}
\end{equation}
Mặc định: patience $= 50$.

\section{Letterbox Resize}

Ảnh được resize về kích thước cố định ($320\times320$ hoặc $640\times640$) bằng letterbox:
\begin{enumerate}
    \item Tính tỷ lệ scale sao cho ảnh fit trong target size
    \item Resize theo tỷ lệ đó (giữ nguyên aspect ratio)
    \item Pad phần dư bằng pixel (114, 114, 114)
\end{enumerate}
